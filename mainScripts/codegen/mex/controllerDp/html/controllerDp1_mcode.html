1function [bestDischargeStep, bestCTG] = controllerDp(cfg, demForecast, ...
  2    pvForecast, battery, hourNow)
  3
  4% controllerDp: Solve dynamic program for the curent horizon, to
  5% minimise costs
  6
  7% Positive b is DISCHARGING the battery!
  8
  9% persistent doneHorizonPlot;
 10
 11%% Initialise Values
 12nStates = length(battery.statesInt);
 13nStages = cfg.sim.horizon;
 14
 15% Initialise array of costs to go (NB: cost from last stage is zero in
 16% all possible states)
 17CTG = zeros(nStates, nStages+1);
 18ST_b = zeros(nStates, nStages);
 19
 20% Limits on battery rate-of-charge:
 21b_min = -battery.maxChargeRate/cfg.sim.stepsPerHour; % kWh/interval
 22b_max = battery.maxChargeRate/cfg.sim.stepsPerHour;  % kWh/interval
 23b_min_int = floor(b_min/battery.increment);       % No. charge increments
 24b_max_int = ceil(b_max/battery.increment);        % No. charge increments
 25
 26% Get the grid prices:
 27importPrices = zeros(nStages,1);
 28exportPrices = zeros(nStages,1);
 29for t = 1:nStages
 30    [importPrices(t), exportPrices(t)] = ...
 31        getGridPrices(cfg, mod(hourNow+t-1, cfg.sim.horizon));
 32end
 33
 34% Work back through previous stages and find minimum cost to go
 35% store the chosen charge decision in ST_b
 36for t = nStages:-1:1
 37    
 38    % For all possible starting states
 39    for q = 1:nStates
 40        
 41        % Initialise bestCTG to large value 
 42        % and best discharge to infeasible value
 43        bestCTG = inf;
 44        bestB = inf;
 45        
 46        % Further constrain minimum and maximum b (charging energy)
 47        this_b_min_int = max(b_min_int, q-nStates);
 48        this_b_max_int = min(b_max_int, q-1);
 49        
 50        % For each feasible discharging decision check the resulting CTG
 51        for thisB = this_b_min_int:this_b_max_int
 52            
 53            % Find net power from battery (account for losses)
 54            if thisB > 0
 55                b_hat = thisB*battery.increment*cfg.sim.batteryEtaD;
 56            else
 57                b_hat = thisB*battery.increment/cfg.sim.batteryEtaC;
 58            end
 59            
 60            % Find energy from grid during interval
 61            g_t = demForecast(t)- b_hat - pvForecast(t);
 62            
 63            % Find battery damage cost
 64            %fracDegradation = calcFracDegradation(cfg, battery, q, ...
 65            %    thisB);
 66            
 67            %damageCost = fracDegradation*battery.Value();
 68            
 69            % Battery degradation cost (fixed per kWh-through-put):
 70            damageCost = abs(thisB)*battery.increment*0.5*...
 71                cfg.bat.costPerKwhUsed;
 72            
 73            % Total state transition cost for this decision from here
 74            thisSTC = importPrices(t)*max(0,g_t) - ...
 75                exportPrices(t)*max(0,-g_t) + damageCost;
 76            
 77            % Total cost-to-got for this decision from here to end
 78            thisCTG = thisSTC + CTG(q-thisB, t+1);
 79            
 80            % Store decision if it's the best found so far
 81            if (thisCTG < bestCTG - cfg.sim.minCostDiff)
 82                bestB = thisB;
 83                bestCTG = thisCTG;
 84            end
 85            
 86        end
 87        
 88        % Store the best discharging decision found
 89        ST_b(q, t) = bestB;
 90        CTG(q, t) = bestCTG;
 91        
 92    end
 93end
 94
 95% Create time-series of charge decisions, and SoC
 96q_t_state = zeros(nStages+1, 1);
 97gridImport = zeros(nStages, 1);
 98q_t_state(1) = battery.state;
 99
100for t=2:nStages
101    % DEBUGGING:
102    % disp('t-1: ');disp(t-1);
103    % disp('q_t_state(t-1): ');disp(q_t_state(t-1));
104    q_t_state(t) = q_t_state(t-1) - ST_b(q_t_state(t-1), t-1);
105    
106    if ST_b(q_t_state(t-1), t-1) > 0
107        b_hat = ST_b(q_t_state(t-1), t-1)*battery.increment*...
108            cfg.sim.batteryEtaD;
109    else
110        b_hat = ST_b(q_t_state(t-1), t-1)*battery.increment/...
111            cfg.sim.batteryEtaC;
112    end
113    gridImport(t) = demForecast(t)- b_hat - pvForecast(t);
114    
115    if(q_t_state(t) < 1 || q_t_state(t)>nStates)
116        error('Battery state out of bounds');
117    end
118end
119
120% Check ending SoC of battery
121q_t_state(nStages+1) = q_t_state(nStages) - ST_b(q_t_state(nStages), ...
122    nStages);
123if(q_t_state(nStages) < 1 || q_t_state(nStages)>nStates)
124    error('Battery state out of bounds');
125end
126
127% Compute the best integer state change of battery
128bestDischargeStep =  ST_b(battery.state, 1);
129bestCTG = CTG(battery.state, 1);
130
131% % DEBUG: Produce plot of optimal horizon decisions for 1st interval:
132% if isempty(doneHorizonPlot)
133%     plotHorizon(cfg, demForecast, pvForecast, q_t_state, hourNow, ...
134%         gridImport);
135%     
136%     doneHorizonPlot = true;
137% end
138
139end
